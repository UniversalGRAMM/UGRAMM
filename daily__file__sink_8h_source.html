26/*
27 * Generator of daily log file names in format basename.YYYY-MM-DD.ext
28 */
29struct daily_filename_calculator {
30 // Create filename for the form basename.YYYY-MM-DD
31 static filename_t calc_filename(const filename_t &filename, const tm &now_tm) {
32 filename_t basename, ext;
33 std::tie(basename, ext) = details::file_helper::split_by_extension(filename);
34 return fmt_lib::format(SPDLOG_FMT_STRING(SPDLOG_FILENAME_T("{}_{:04d}-{:02d}-{:02d}{}")),
35 basename, now_tm.tm_year + 1900, now_tm.tm_mon + 1, now_tm.tm_mday,
36 ext);
37 }
38};

40/*
41 * Generator of daily log file names with strftime format.
42 * Usages:
43 * auto sink =
44 * std::make_shared<spdlog::sinks::daily_file_format_sink_mt>("myapp-%Y-%m-%d:%H:%M:%S.log", hour,
45 * minute);" auto logger = spdlog::daily_logger_format_mt("loggername, "myapp-%Y-%m-%d:%X.log",
46 * hour, minute)"
47 *
48 */
49struct daily_filename_format_calculator {
50 static filename_t calc_filename(const filename_t &file_path, const tm &now_tm) {
51#if defined(_WIN32) && defined(SPDLOG_WCHAR_FILENAMES)
52 std::wstringstream stream;
53#else
54 std::stringstream stream;
55#endif
56 stream << std::put_time(&now_tm, file_path.c_str());
57 return stream.str();
58 }
59};

61/*
62 * Rotating file sink based on date.
63 * If truncate != false , the created file will be truncated.
64 * If max_files > 0, retain only the last max_files and delete previous.
65 */
66template <typename Mutex, typename FileNameCalc = daily_filename_calculator>
67class daily_file_sink final : public base_sink<Mutex> {
68public:
69 // create daily file sink which rotates on given time
70 daily_file_sink(filename_t base_filename,
71 int rotation_hour,
72 int rotation_minute,
73 bool truncate = false,
74 uint16_t max_files = 0,
75 const file_event_handlers &event_handlers = {})
76 : base_filename_(std::move(base_filename)),
77 rotation_h_(rotation_hour),
78 rotation_m_(rotation_minute),
79 file_helper_{event_handlers},
80 truncate_(truncate),
81 max_files_(max_files),
82 filenames_q_() {
83 if (rotation_hour < 0 || rotation_hour > 23 || rotation_minute < 0 ||
84 rotation_minute > 59) {
85 throw_spdlog_ex("daily_file_sink: Invalid rotation time in ctor");
86 }
87
88 auto now = log_clock::now();
89 auto filename = FileNameCalc::calc_filename(base_filename_, now_tm(now));
90 file_helper_.open(filename, truncate_);
91 rotation_tp_ = next_rotation_tp_();
92
93 if (max_files_ > 0) {
94 init_filenames_q_();
95 }
96 }
97
98 filename_t filename() {
99 std::lock_guard<Mutex> lock(base_sink<Mutex>::mutex_);
100 return file_helper_.filename();
101 }
102
103protected:
104 void sink_it_(const details::log_msg &msg) override {
105 auto time = msg.time;
106 bool should_rotate = time >= rotation_tp_;
107 if (should_rotate) {
108 auto filename = FileNameCalc::calc_filename(base_filename_, now_tm(time));
109 file_helper_.open(filename, truncate_);
110 rotation_tp_ = next_rotation_tp_();
111 }
112 memory_buf_t formatted;
113 base_sink<Mutex>::formatter_->format(msg, formatted);
114 file_helper_.write(formatted);
115
116 // Do the cleaning only at the end because it might throw on failure.
117 if (should_rotate && max_files_ > 0) {
118 delete_old_();
119 }
120 }
121
122 void flush_() override { file_helper_.flush(); }
123
124private:
125 void init_filenames_q_() {
126 using details::os::path_exists;
127
128 filenames_q_ = details::circular_q<filename_t>(static_cast<size_t>(max_files_));
129 std::vector<filename_t> filenames;
130 auto now = log_clock::now();
131 while (filenames.size() < max_files_) {
132 auto filename = FileNameCalc::calc_filename(base_filename_, now_tm(now));
133 if (!path_exists(filename)) {
134 break;
135 }
136 filenames.emplace_back(filename);
137 now -= std::chrono::hours(24);
138 }
139 for (auto iter = filenames.rbegin(); iter != filenames.rend(); ++iter) {
140 filenames_q_.push_back(std::move(*iter));
141 }
142 }
143
144 tm now_tm(log_clock::time_point tp) {
145 time_t tnow = log_clock::to_time_t(tp);
146 return spdlog::details::os::localtime(tnow);
147 }
148
149 log_clock::time_point next_rotation_tp_() {
150 auto now = log_clock::now();
151 tm date = now_tm(now);
152 date.tm_hour = rotation_h_;
153 date.tm_min = rotation_m_;
154 date.tm_sec = 0;
155 auto rotation_time = log_clock::from_time_t(std::mktime(&date));
156 if (rotation_time > now) {
157 return rotation_time;
158 }
159 return {rotation_time + std::chrono::hours(24)};
160 }
161
162 // Delete the file N rotations ago.
163 // Throw spdlog_ex on failure to delete the old file.
164 void delete_old_() {
165 using details::os::filename_to_str;
166 using details::os::remove_if_exists;
167
168 filename_t current_file = file_helper_.filename();
169 if (filenames_q_.full()) {
170 auto old_filename = std::move(filenames_q_.front());
171 filenames_q_.pop_front();
172 bool ok = remove_if_exists(old_filename) == 0;
173 if (!ok) {
174 filenames_q_.push_back(std::move(current_file));
175 throw_spdlog_ex("Failed removing daily file " + filename_to_str(old_filename),
176 errno);
177 }
178 }
179 filenames_q_.push_back(std::move(current_file));
180 }
181
182 filename_t base_filename_;
183 int rotation_h_;
184 int rotation_m_;
185 log_clock::time_point rotation_tp_;
186 details::file_helper file_helper_;
187 bool truncate_;
188 uint16_t max_files_;
189 details::circular_q<filename_t> filenames_q_;
190};
